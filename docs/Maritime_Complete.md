/* ───────────────────────────────────────────────────────────────────────────────
 * Core μ Calculation — Weighted Geometric Mean with Domain Coherence
 * ─────────────────────────────────────────────────────────────────────────────── */

float calculate_harmony_mu(MaritimeContext* ctx) {
    float domain_log_sums[DOMAIN_COUNT] = {0.0f};
    float domain_weight_sums[DOMAIN_COUNT] = {0.0f};

    /* Aggregate subsystem scores by constitutional domain */
    for (uint8_t i = 0; i < ctx->count; i++) {
        float score = ctx->subsystems[i].score;
        if (score < MIN_SCORE) score = MIN_SCORE;
        if (score > 1.0f) score = 1.0f;

        uint8_t d = ctx->subsystems[i].domain;
        domain_log_sums[d] += ctx->subsystems[i].weight * logf(score);
        domain_weight_sums[d] += ctx->subsystems[i].weight;
    }

    /* Normalize and compute domain-level μ */
    float total_log_sum = 0.0f;
    for (uint8_t d = 0; d < DOMAIN_COUNT; d++) {
        if (domain_weight_sums[d] > 0.0f) {
            ctx->domain_mu[d] = expf(domain_log_sums[d] / domain_weight_sums[d]);
            total_log_sum += domain_weights[d] * logf(ctx->domain_mu[d]);
        } else {
            ctx->domain_mu[d] = 1.0f;
        }
    }

    ctx->total_mu = expf(total_log_sum);
    return ctx->total_mu;
}

/* Hull structural harmony — μ_mechanical */
float calculate_hull_harmony(MaritimeContext* ctx) {
    /* Watertight integrity */
    float wt_integrity = 1.0f;
    uint8_t compromised_compartments = 0;

    for (uint8_t i = 0; i < ctx->compartment_count; i++) {
        WatertightCompartment* comp = &ctx->compartments[i];
        if (comp->flooding_detected || comp->bilge_alarm_active) {
            wt_integrity *= 0.7f;
            compromised_compartments++;
        }
        if (!comp->watertight_door_closed && comp->bilge_level_mm > 50.0f) {
            wt_integrity *= 0.5f;
        }
        /* Structural stress */
        float stress_score = expf(-comp->structural_stress_mpa / 200.0f); /* 200MPa yield */
        wt_integrity *= stress_score;
    }

    /* List and trim */
    float list_score = expf(-fabsf(ctx->motion.list_deg) / 5.0f);    /* 5° limit */
    float trim_score = expf(-fabsf(ctx->motion.trim_m) / 2.0f);      /* 2m limit */

    /* Draft within load line */
    float draft_score = 1.0f;
    if (ctx->motion.draft_m > 20.0f) { /* Summer load line proxy */
        draft_score = expf(-(ctx->motion.draft_m - 20.0f) / 2.0f);
    }

    /* Roll stability */
    float roll_score = expf(-fabsf(ctx->motion.roll_deg) / 30.0f); /* 30° limit */

    float hull_mu = wt_integrity * list_score * trim_score * draft_score * roll_score;
    return (hull_mu < MIN_SCORE) ? MIN_SCORE : hull_mu;
}

/* Navigation harmony — μ_ionic */
float calculate_navigation_harmony(MaritimeContext* ctx) {
    /* CPA/TCPA collision avoidance */
    float collision_score = 1.0f;
    if (ctx->nav.collision_risk) {
        collision_score = expf(-1.0f / (ctx->nav.cpa_nm + 0.1f)) *
                         expf(-ctx->nav.tcpa_min / 30.0f); /* 30min threshold */
    }

    /* Grounding avoidance */
    float grounding_score = 1.0f;
    if (ctx->nav.grounding_risk) {
        grounding_score = expf(-ctx->nav.distance_to_shore_nm / 2.0f) *
                          expf(-ctx->motion.depth_below_keel_m / 5.0f);
    }

    /* Piracy zone */
    float piracy_score = 1.0f;
    if (ctx->nav.in_piracy_zone) {
        piracy_score = 0.7f; /* Elevated risk, not automatic prohibition */
    }

    /* Ice navigation */
    float ice_score = 1.0f;
    if (ctx->nav.in_ice) {
        ice_score = 0.8f; /* Ice class dependent */
    }

    /* TSS compliance */
    float tss_score = 1.0f;
    if (ctx->nav.in_tss) {
        tss_score = 0.95f; /* Minor penalty for complexity */
    }

    float nav_mu = collision_score * grounding_score * piracy_score * ice_score * tss_score;
    return (nav_mu < MIN_SCORE) ? MIN_SCORE : nav_mu;
}

/* Cargo harmony — μ_chemical + μ_mechanical */
float calculate_cargo_harmony(MaritimeContext* ctx) {
    if (ctx->hold_count == 0) return 1.0f;

    float log_sum = 0.0f;
    float weight_sum = 0.0f;

    for (uint8_t i = 0; i < ctx->hold_count; i++) {
        CargoHold* hold = &ctx->holds[i];

        /* Temperature control (reefer) */
        float temp_score = 1.0f;
        if (hold->reefer_active) {
            temp_score = expf(-fabsf(hold->cargo_temperature_c - (-18.0f)) / 5.0f); /* Frozen proxy */
        }

        /* Humidity */
        float humidity_score = expf(-fabsf(hold->humidity_percent - 65.0f) / 20.0f);

        /* Lashing integrity */
        float lashing = hold->lashing_integrity;

        /* Hazardous cargo safety */
        float hazardous_score = 1.0f;
        if (hold->hazardous) {
            hazardous_score = 0.9f; /* Slight penalty, managed by IMDG */
        }

        /* Fumigation safety */
        float fumigation = hold->fumigated ? 0.95f : 1.0f;

        float hold_score = temp_score * humidity_score * lashing * hazardous_score * fumigation;
        hold_score = (hold_score < MIN_SCORE) ? MIN_SCORE : hold_score;

        float weight = hold->cargo_weight_t / 1000.0f;
        log_sum += weight * logf(hold_score);
        weight_sum += weight;
    }

    return (weight_sum > 0.0f) ? expf(log_sum / weight_sum) : 1.0f;
}

/* Ballast water harmony — μ_chemical environmental */
float calculate_ballast_harmony(MaritimeContext* ctx) {
    BallastWater* ballast = &ctx->ballast;

    /* Treatment system health */
    float treatment = ballast->treatment_system_health;

    /* Exchange completion */
    float exchange = ballast->exchange_completed ? 1.0f : 0.5f;

    /* Treatment completion */
    float treatment_done = ballast->treatment_completed ? 1.0f : 0.5f;

    /* Invasive species detection */
    float invasive = ballast->invasive_species_detected ? 0.1f : 1.0f;

    /* Salinity appropriate for destination */
    float salinity = expf(-fabsf(ballast->salinity_ppt - 35.0f) / 10.0f);

    float ballast_mu = treatment * exchange * treatment_done * invasive * salinity;
    return (ballast_mu < MIN_SCORE) ? MIN_SCORE : ballast_mu;
}

/* Engine harmony — μ_thermal + μ_mechanical */
float calculate_engine_harmony(MaritimeContext* ctx) {
    EngineState* eng = &ctx->engine;

    /* Load within design */
    float load_score = expf(-fabsf(eng->main_engine_load_percent - 85.0f) / 15.0f);

    /* Exhaust temperature */
    float exhaust = expf(-fabsf(eng->exhaust_temp_c - 400.0f) / 50.0f);

    /* Scavenge pressure */
    float scavenge = expf(-fabsf(eng->scavenge_pressure_kpa - 2.5f) / 0.5f);

    /* Turbocharger speed */
    float turbo = expf(-fabsf(eng->turbocharger_speed_rpm - 15000.0f) / 3000.0f);

    /* Crankcase pressure (negative = good) */
    float crankcase = expf(-eng->crankcase_pressure_kpa / 0.5f);

    /* Slow steaming efficiency */
    float slow_steam = eng->slow_steam_active ? 0.9f : 1.0f;

    float engine_mu = load_score * exhaust * scavenge * turbo * crankcase * slow_steam;
    return (engine_mu < MIN_SCORE) ? MIN_SCORE : engine_mu;
}

/* Emissions harmony — μ_valley environmental */
float calculate_emissions_harmony(MaritimeContext* ctx) {
    EmissionsState* em = &ctx->emissions;

    /* SOx compliance (IMO 2020: 0.5% global, 0.1% ECA) */
    float sox = expf(-em->sox_ppm / 50.0f); /* 50ppm proxy */

    /* NOx compliance (Tier III in ECA) */
    float nox = expf(-em->nox_ppm / 100.0f);

    /* CO2 intensity */
    float co2 = expf(-em->co2_kg_nm / 50.0f); /* 50kg/nm benchmark */

    /* Particulate */
    float particulate = expf(-em->particulate_mg_nm3 / 50.0f);

    /* Methane slip (LNG) */
    float methane = expf(-em->methane_slip_percent / 3.0f); /* 3% limit */

    /* CII rating */
    float cii = 1.0f;
    if (em->cii_rating > 3.0f) { /* D or E */
        cii = 0.1f;
    }

    /* Carbon budget */
    float budget = (em->carbon_budget_remaining_tons > 0.0f) ? 1.0f : 0.1f;

    float emissions_mu = sox * nox * co2 * particulate * methane * cii * budget;
    return (emissions_mu < MIN_SCORE) ? MIN_SCORE : emissions_mu;
}

    float nav_mu = calculate_navigation_harmony(ctx);
    ctx->domain_mu[1] = (ctx->domain_mu[1] + nav_mu) / 2.0f; /* μ_ionic */

    float cargo_mu = calculate_cargo_harmony(ctx);
    ctx->domain_mu[0] = (ctx->domain_mu[0] + cargo_mu) / 2.0f; /* μ_chemical */

    float ballast_mu = calculate_ballast_harmony(ctx);
    ctx->domain_mu[0] = (ctx->domain_mu[0] + ballast_mu) / 2.0f;

    float engine_mu = calculate_engine_harmony(ctx);
    ctx->domain_mu[3] = (ctx->domain_mu[3] + engine_mu) / 2.0f; /* μ_thermal */

    float emissions_mu = calculate_emissions_harmony(ctx);
    ctx->domain_mu[5] = (ctx->domain_mu[5] + emissions_mu) / 2.0f; /* μ_valley */

    return (
        /* μ_law1: Zero Exception */
        ctx->domain_mu[0] > 0.9990f && /* chemical — cargo/ballast */
        ctx->domain_mu[1] > 0.9990f && /* ionic — navigation */
        ctx->domain_mu[4] > 0.9990f && /* mechanical — hull */

        /* μ_law3: No Bypass — departure authorized */
        (ctx->voyage_phase > 0 || ctx->departure_authorized),

        /* μ_law7: No Silent Failure */
        ctx->subsystems[0].health_flag && /* radar */
        ctx->subsystems[1].health_flag && /* AIS */
        ctx->subsystems[2].health_flag,   /* ECDIS */

        /* μ_law9: Sovereign Identity — IMO verified */
        ctx->imo_number[0] != '\0',

        /* μ_law11: Integrity Preservation */
        ctx->total_mu > 0.9997f,

        /* μ_law12: No man overboard */
        !ctx->man_overboard_alert,

        /* μ_law15: Energy Accounting — emissions budget */
        ctx->emissions.carbon_budget_remaining_tons > 0.0f,

        /* μ_law16: Cross-Domain Consistency */
        ctx->domain_mu[5] > 0.9990f,

        /* No abandon ship */
        !ctx->abandon_ship_ordered
    );
}

/* ───────────────────────────────────────────────────────────────────────────────
 * Maritime Decision Gate
 * ─────────────────────────────────────────────────────────────────────────────── */

typedef enum {
    VOYAGE_GO       = 0x474F,  /* "GO" */
    VOYAGE_HALT     = 0x4841,  /* "HA"lt */
    VOYAGE_HEAL     = 0x4845,  /* "HE"al */
    VOYAGE_EMERGENCY= 0x454D,  /* "EM"ergency */
    VOYAGE_DIVERT   = 0x4449,  /* "DI"vert */
    VOYAGE_ABANDON  = 0x4142   /* "AB"andon */
} MaritimeDecision;

MaritimeDecision evaluate_maritime_harmony(MaritimeContext* ctx) {
    float mu = calculate_harmony_mu(ctx);
    bool ch = check_codified_harmony(ctx);

    ctx->ch_state = ch;
    ctx->cycle_count++;

    if (mu >= HARMONY_THRESHOLD && ch && !ctx->man_overboard_alert) {
        return VOYAGE_GO;
    } else if (mu >= 0.9990f && ch) {
        trigger_healing_sequence(ctx);
        return VOYAGE_HEAL;
    } else if (mu >= 0.9980f && !ch) {
        return VOYAGE_HALT;
    } else if (ctx->nav.grounding_risk || ctx->nav.collision_risk) {
        return VOYAGE_DIVERT;
    } else {
        trigger_emergency_sequence(ctx);
        log_harmony_fault(ctx, mu, ch);
        return (mu < 0.9950f) ? VOYAGE_ABANDON : VOYAGE_EMERGENCY;
    }
}

/* ───────────────────────────────────────────────────────────────────────────────
 * Healing Sequence — μ_law4: Self-Healing
 * ─────────────────────────────────────────────────────────────────────────────── */

void trigger_healing_sequence(MaritimeContext* ctx) {
    /* Weather routing adjustment */
    if (ctx->motion.sog_kts > 15.0f && ctx->nav.in_piracy_zone) {
        ctx->engine.slow_steam_active = true;
        reduce_speed(ctx, 12.0f);
    }

    /* Ballast adjustment on list */
    if (fabsf(ctx->motion.list_deg) > 2.0f) {
        transfer_ballast(ctx);
    }

    /* Engine derating on thermal stress */
    if (ctx->engine.exhaust_temp_c > 450.0f) {
        derate_engine(ctx, 0.8f);
    }

    /* Anti-fouling on performance degradation */
    if (ctx->engine.fuel_consumption_kg_h > ctx->engine.main_engine_load_percent * 2.0f) {
        activate_anti_fouling(ctx);
    }

    /* Course alteration on collision risk */
    if (ctx->nav.collision_risk && ctx->nav.tcpa_min < 30.0f) {
        alter_course(ctx, 20.0f);
    }
}

/* ───────────────────────────────────────────────────────────────────────────────
 * Emergency Sequence — μ_law14: Finite State Guarantee
 * ─────────────────────────────────────────────────────────────────────────────── */

void trigger_emergency_sequence(MaritimeContext* ctx) {
    /* Engine stop */
    stop_engine(ctx);

    /* MAYDAY if man overboard or flooding */
    if (ctx->man_overboard_alert || ctx->compartments[0].flooding_detected) {
        broadcast_mayday(ctx);
        activate_epirb(ctx);
    }

    /* Anchor if grounding risk */
    if (ctx->nav.grounding_risk && ctx->motion.depth_below_keel_m < 20.0f) {
        emergency_anchor(ctx);
    }

    /* Watertight closure */
    for (uint8_t i = 0; i < ctx->compartment_count; i++) {
        if (ctx->compartments[i].flooding_detected) {
            ctx->compartments[i].watertight_door_closed = true;
            close_watertight_door(&ctx->compartments[i]);
        }
    }

    /* Crew muster */
    sound_general_alarm(ctx);

    /* Abandon ship if critical */
    if (ctx->total_mu < 0.9900f) {
        ctx->abandon_ship_ordered = true;
        order_abandon_ship(ctx);
    }
}

#define MIN_SCORE 1e-12f
#define MAX_SUBSYSTEMS 8
#define DOMAIN_COUNT 6

class HarmonyMaritime_v2 {
private:
    float scores[MAX_SUBSYSTEMS];
    float weights[MAX_SUBSYSTEMS];
    byte domains[MAX_SUBSYSTEMS];
    byte count;
    float domainMu[DOMAIN_COUNT];
    float sog_kts;
    float cpa_nm;
    float depth_below_keel_m;
    bool departureAuthorized;
    bool manOverboard;

public:
    HarmonyMaritime_v2() : count(0), sog_kts(15.0f), cpa_nm(5.0f),
        depth_below_keel_m(50.0f), departureAuthorized(false),
        manOverboard(false) {
        memset(scores, 0, sizeof(scores));
        memset(weights, 0, sizeof(weights));
        memset(domains, 0, sizeof(domains));
        memset(domainMu, 0, sizeof(domainMu));
    }

    void addSubsystem(float weight, float initialScore, byte domain) {
        if (count < MAX_SUBSYSTEMS && domain < DOMAIN_COUNT) {
            weights[count] = weight;
            scores[count] = constrain(initialScore, 0.0001f, 1.0f);
            domains[count] = domain;
            count++;
        }
    }

    float computeMu() {
        float domainLogSums[DOMAIN_COUNT] = {0.0f};
        float domainWeightSums[DOMAIN_COUNT] = {0.0f};

        for (byte i = 0; i < count; i++) {
            float s = scores[i];
            if (s < MIN_SCORE) s = MIN_SCORE;
            byte d = domains[i];
            domainLogSums[d] += weights[i] * log(s);
            domainWeightSums[d] += weights[i];
        }

        float totalLogSum = 0.0f;
        const float domainWeights[DOMAIN_COUNT] = {0.15f, 0.20f, 0.10f,
                                                    0.15f, 0.20f, 0.20f};

        for (byte d = 0; d < DOMAIN_COUNT; d++) {
            if (domainWeightSums[d] > 0.0f) {
                domainMu[d] = exp(domainLogSums[d] / domainWeightSums[d]);
                totalLogSum += domainWeights[d] * log(domainMu[d]);
            } else {
                domainMu[d] = 1.0f;
            }
        }

        return exp(totalLogSum);
    }

    bool checkCH() {
        /* μ_law1: Zero Exception */
        bool ionicOK = domainMu[1] > 0.9990f;
        bool mechanicalOK = domainMu[4] > 0.9990f;

        /* μ_law3: No Bypass — departure authorized */
        bool departureOK = departureAuthorized || sog_kts < 1.0f;

        /* μ_law7: No Silent Failure */
        bool responsive = true;

        /* μ_law9: Sovereign Identity — depth safe */
        bool depthOK = depth_below_keel_m > 10.0f;

        /* μ_law11: Integrity Preservation */
        float totalMu = computeMu();
        bool marginOK = totalMu > 0.9997f;

        /* μ_law12: CPA safe */
        bool cpaOK = cpa_nm > 1.0f;

        /* μ_law16: Cross-Domain Consistency — no MOB */
        bool mobOK = !manOverboard;

        return ionicOK && mechanicalOK && departureOK && responsive &&
               depthOK && marginOK && cpaOK && mobOK;
    }

    bool isHarmonious() {
        return computeMu() >= HARMONY_THRESHOLD && checkCH();
    }

    void updateScore(byte index, float newScore) {
        if (index < count) {
            scores[index] = constrain(newScore, 0.0001f, 1.0f);
        }
    }

    void setSOG(float kts) { sog_kts = constrain(kts, 0.0f, 30.0f); }
    void setCPA(float nm) { cpa_nm = constrain(nm, 0.0f, 20.0f); }
    void setDepth(float m) { depth_below_keel_m = constrain(m, 0.0f, 1000.0f); }
    void setDepartureAuthorized(bool auth) { departureAuthorized = auth; }
    void setManOverboard(bool mob) { manOverboard = mob; }
    float getDomainMu(byte domain) { return (domain < DOMAIN_COUNT) ? domainMu[domain] : 0.0f; }
    float getSOG() { return sog_kts; }
    float getCPA() { return cpa_nm; }
};
