float calculate_harmony_mu(NuclearContext* ctx) {
    float domain_log_sums[DOMAIN_COUNT] = {0.0f};
    float domain_weight_sums[DOMAIN_COUNT] = {0.0f};

    for (uint8_t i = 0; i < ctx->count; i++) {
        float score = ctx->subsystems[i].score;
        if (score < MIN_SCORE) score = MIN_SCORE;
        if (score > 1.0f) score = 1.0f;

        uint8_t d = ctx->subsystems[i].domain;
        domain_log_sums[d] += ctx->subsystems[i].weight * logf(score);
        domain_weight_sums[d] += ctx->subsystems[i].weight;
    }

    float total_log_sum = 0.0f;
    for (uint8_t d = 0; d < DOMAIN_COUNT; d++) {
        if (domain_weight_sums[d] > 0.0f) {
            ctx->domain_mu[d] = expf(domain_log_sums[d] / domain_weight_sums[d]);
            total_log_sum += domain_weights[d] * logf(ctx->domain_mu[d]);
        } else {
            ctx->domain_mu[d] = 1.0f;
        }
    }

    ctx->total_mu = expf(total_log_sum);
    return ctx->total_mu;
}

/* Fuel integrity harmony — μ_chemical + μ_mechanical */
float calculate_fuel_harmony(NuclearContext* ctx) {
    if (ctx->fuel_count == 0) return 1.0f;

    float log_sum = 0.0f;
    float weight_sum = 0.0f;

    for (uint16_t i = 0; i < ctx->fuel_count; i++) {
        FuelAssembly* fa = &ctx->fuel[i];

        /* Cladding integrity */
        float cladding = fa->cladding_integrity;

        /* Burnup limit — 62 MWD/kg typical PWR limit */
        float burnup_score = expf(-fmaxf(0.0f, fa->burnup_mwd_kg - 55.0f) / 7.0f);

        /* Fuel temperature limit */
        float temp_score = (fa->fuel_temperature_c < 2200.0f) ? 1.0f : 0.1f;

        /* Failed fuel penalty */
        float failed_penalty = fa->failed_fuel_flag ? 0.1f : 1.0f;
        float leaker_penalty = fa->leaker_flag ? 0.5f : 1.0f;

        /* Local power fraction within design */
        float power_score = (fa->local_power_fraction < 1.5f) ? 1.0f : 0.5f;

        float assembly_score = cladding * burnup_score * temp_score *
                              failed_penalty * leaker_penalty * power_score;
        assembly_score = (assembly_score < MIN_SCORE) ? MIN_SCORE : assembly_score;

        float weight = fa->local_power_fraction;
        log_sum += weight * logf(assembly_score);
        weight_sum += weight;
    }

    return (weight_sum > 0.0f) ? expf(log_sum / weight_sum) : 1.0f;
}

/* Control rod harmony — μ_mechanical + μ_quantum */
float calculate_rod_harmony(NuclearContext* ctx) {

    for (uint8_t i = 0; i < ctx->rod_count; i++) {
        ControlRod* rod = &ctx->rods[i];

        /* Position indication accuracy */

        /* Drive mechanism health */
        float drive_health = rod->drive_motor_health;

        /* Stuck rod penalty */
        float stuck_penalty = rod->stuck_rod_alarm ? 0.1f : 1.0f;

        /* Scram capability */

        float weight = rod->worth_pcm / 1000.0f;
        log_sum += weight * logf(rod_score);
        weight_sum += weight;
    }

    return (weight_sum > 0.0f) ? expf(log_sum / weight_sum) : 1.0f;
}

/* Core thermal-hydraulic harmony — μ_thermal (dominant) */
float calculate_thermal_harmony(NuclearContext* ctx) {
    CoreThermalHydraulics* th = &ctx->core_th;

    /* DNBR — must be > 1.3 for safety */
    float dnbr_score = (th->dnbr > 1.5f) ? 1.0f :
                       (th->dnbr > 1.3f) ? expf(-(1.5f - th->dnbr) / 0.2f) : 0.1f;

    /* Cladding surface temperature — limit 343°C for Zircaloy */
    float cladding_score = (th->cladding_surface_temp_c < 330.0f) ? 1.0f :
                           expf(-(th->cladding_surface_temp_c - 330.0f) / 13.0f);

    /* Peak fuel centerline — limit ~2200°C */
    float centerline_score = (th->peak_fuel_centerline_temp_c < 2000.0f) ? 1.0f :
                             expf(-(th->peak_fuel_centerline_temp_c - 2000.0f) / 200.0f);

    /* Coolant pressure within design (~15.5 MPa for PWR) */
    float pressure_score = (th->coolant_pressure_mpa > 14.0f && th->coolant_pressure_mpa < 17.0f) ?
                           1.0f : 0.5f;

    /* Flow rate adequate */
    float flow_score = (th->coolant_flow_rate_kg_s > 10000.0f) ? 1.0f : 0.7f;

    /* Void fraction acceptable */
    float void_score = (th->void_fraction_core_avg < 0.4f) ? 1.0f :
                       expf(-(th->void_fraction_core_avg - 0.4f) / 0.1f);

    float thermal_mu = dnbr_score * cladding_score * centerline_score *
                      pressure_score * flow_score * void_score;
    return (thermal_mu < MIN_SCORE) ? MIN_SCORE : thermal_mu;
}

    float rps_mu = (weight_sum > 0.0f) ? expf(log_sum / weight_sum) * redundancy_penalty : 1.0f;
    return (rps_mu < MIN_SCORE) ? MIN_SCORE : rps_mu;

/* ───────────────────────────────────────────────────────────────────────────────
 * Nuclear Decision Gate
 * ─────────────────────────────────────────────────────────────────────────────── */

typedef enum {
    REACTOR_GO      = 0x474F,  /* "GO" */
    REACTOR_HALT    = 0x4841,  /* "HA"lt */
    REACTOR_HEAL    = 0x4845,  /* "HE"al */
    REACTOR_SCRAM   = 0x5343,  /* "SC"ram */
    REACTOR_SI      = 0x5349,  /* "SI" — safety injection */
    REACTOR_ECAS    = 0x4543,  /* "EC" — emergency core cooling */
    REACTOR_FSAR    = 0x4653   /* "FS" — FSAR emergency procedure */
} NuclearDecision;

NuclearDecision evaluate_nuclear_harmony(NuclearContext* ctx) {
    float mu = calculate_harmony_mu(ctx);
    bool ch = check_codified_harmony(ctx);

    ctx->ch_state = ch;
    ctx->cycle_count++;

    if (mu >= HARMONY_THRESHOLD && ch && !ctx->reactor_trip_status &&
        ctx->reactor_power_percent > 0.0f) {
        return REACTOR_GO;
    } else if (mu >= 0.9990f && ch && !ctx->reactor_trip_status) {
        trigger_healing_sequence(ctx);
        return REACTOR_HEAL;
    } else if (mu >= 0.9980f && !ch && !ctx->reactor_trip_status) {
        return REACTOR_HALT;
    } else if (ctx->core_th.dnbr < 1.3f || ctx->core_th.cladding_surface_temp_c > 343.0f) {
        trigger_scram(ctx);
        return REACTOR_SCRAM;
    } else if (ctx->core_th.coolant_pressure_mpa < 12.0f || ctx->containment.pressure_kpa > 500.0f) {
        trigger_safety_injection(ctx);
        return REACTOR_SI;
    } else {
        trigger_emergency_procedure(ctx);
        log_harmony_fault(ctx, mu, ch);
        return (mu < 0.9900f) ? REACTOR_FSAR : REACTOR_ECAS;
    }
}

/* ───────────────────────────────────────────────────────────────────────────────
 * Healing Sequence — μ_law4: Self-Healing
 * ─────────────────────────────────────────────────────────────────────────────── */

void trigger_healing_sequence(NuclearContext* ctx) {
    /* Boron injection if reactivity rising */
    if (ctx->reactor_power_percent > 100.0f && !ctx->reactor_trip_status) {
        inject_boron(ctx, 100.0f);  /* ppm */
    }

    /* Auxiliary feedwater if SG level dropping */
    if (ctx->subsystems[2].score < 0.9995f) {
        actuate_auxiliary_feedwater(ctx);
    }

    /* Containment spray if pressure rising */
    if (ctx->containment.pressure_kpa > 150.0f) {
        actuate_containment_spray(ctx);
    }

    /* SFP cooling if temperature rising */
    if (ctx->sfp.water_temperature_c > 45.0f) {
        augment_sfp_cooling(ctx);
    }

    /* Grid frequency support */
    if (ctx->grid_connected && (ctx->grid_frequency_hz < 59.8f || ctx->grid_frequency_hz > 60.2f)) {
        adjust_load(ctx);
    }
}

    ctx->reactor_power_percent = 0.0f;
}

    /* Safety injection signal */
    ctx->si_signal_active = true;

    /* ECCS actuation */
    actuate_hpi(ctx);   /* High pressure injection */
    actuate_lpci(ctx);  /* Low pressure coolant injection */

    isolate_containment(ctx);

    /* Auxiliary feedwater */
    ctx->auxiliary_feed_actuated = true;
    actuate_auxiliary_feedwater(ctx);

    /* Emergency diesel */
    if (!ctx->grid_connected) {
        ctx->emergency_diesel_running = true;
        start_emergency_diesel(ctx);
    }
}

void trigger_emergency_procedure(NuclearContext* ctx) {
    /* Reactor trip if not already */

    trip_turbine(ctx);

}

    ctx->subsystems[0].last_score = ctx->subsystems[0].score;
    ctx->subsystems[0].score = read_neutron_flux();

    /* Update core thermal-hydraulics */
    update_core_th(ctx);

    /* Update RPS channels */
    update_rps_channels(ctx);

    /* Update containment */
    update_containment(ctx);

    /* Update spent fuel pool */
    update_sfp(ctx);

    /* Update safeguards */
    update_safeguards(ctx);

    /* Update grid */
    update_grid_status(ctx);

    /* Evaluate decision */
    NuclearDecision decision = evaluate_nuclear_harmony(ctx);

    switch (decision) {
        case REACTOR_GO:
            execute_power_operation(ctx);
            break;
        case REACTOR_HEAL:
            execute_healing_mode(ctx);
            break;
        case REACTOR_HALT:
            enter_hot_standby(ctx);
            break;
        case REACTOR_SCRAM:
            execute_scram_procedure(ctx);
            break;
        case REACTOR_SI:
            execute_safety_injection_procedure(ctx);
            break;
        case REACTOR_ECAS:
            execute_emergency_core_cooling(ctx);
            break;
        case REACTOR_FSAR:
            execute_fsar_emergency_procedure(ctx);
            break;
    }
}

#define MAX_SUBSYSTEMS 8
#define DOMAIN_COUNT 6

class HarmonyNuclear_v2 {
private:
    float scores[MAX_SUBSYSTEMS];
    float weights[MAX_SUBSYSTEMS];
    byte domains[MAX_SUBSYSTEMS];
    byte count;
    float domainMu[DOMAIN_COUNT];
    float reactorPowerPercent;
    float dnbr;
    float containmentPressureKpa;
    bool reactorTripStatus;
    bool siSignalActive;

public:
    HarmonyNuclear_v2() : count(0), reactorPowerPercent(100.0f), dnbr(2.5f),
        containmentPressureKpa(101.3f), reactorTripStatus(false),
        siSignalActive(false) {
        memset(scores, 0, sizeof(scores));
        memset(weights, 0, sizeof(weights));
        memset(domains, 0, sizeof(domains));
        memset(domainMu, 0, sizeof(domainMu));
    }

    void addSubsystem(float weight, float initialScore, byte domain) {
        if (count < MAX_SUBSYSTEMS && domain < DOMAIN_COUNT) {
            weights[count] = weight;
            scores[count] = constrain(initialScore, 0.0001f, 1.0f);
            domains[count] = domain;
            count++;
        }
    }

    float computeMu() {
        float domainLogSums[DOMAIN_COUNT] = {0.0f};
        float domainWeightSums[DOMAIN_COUNT] = {0.0f};

        for (byte i = 0; i < count; i++) {
            float s = scores[i];
            if (s < MIN_SCORE) s = MIN_SCORE;
            byte d = domains[i];
            domainLogSums[d] += weights[i] * log(s);
            domainWeightSums[d] += weights[i];
        }

        float totalLogSum = 0.0f;
        const float domainWeights[DOMAIN_COUNT] = {0.15f, 0.20f, 0.15f,
                                                    0.20f, 0.15f, 0.15f};

        for (byte d = 0; d < DOMAIN_COUNT; d++) {
            if (domainWeightSums[d] > 0.0f) {
                domainMu[d] = exp(domainLogSums[d] / domainWeightSums[d]);
                totalLogSum += domainWeights[d] * log(domainMu[d]);
            } else {
                domainMu[d] = 1.0f;
            }
        }

        return exp(totalLogSum);
    }

    bool checkCH() {
        /* μ_law1: Zero Exception */
        bool chemicalOK = domainMu[0] > 0.9990f;
        bool ionicOK = domainMu[1] > 0.9990f;
        bool thermalOK = domainMu[3] > 0.9990f;
        bool mechanicalOK = domainMu[4] > 0.9990f;

        /* μ_law3: No Bypass — reactor trip available */
        bool tripOK = !reactorTripStatus || reactorPowerPercent < 5.0f;

        /* μ_law7: No Silent Failure */
        bool responsive = true;

        /* μ_law9: Sovereign Identity — DNBR valid */
        bool dnbrOK = dnbr > 1.5f;

        /* μ_law11: Integrity Preservation */
        float totalMu = computeMu();
        bool marginOK = totalMu > 0.9997f;

        /* μ_law12: Containment integrity */
        bool containmentOK = containmentPressureKpa < 500.0f;

        /* μ_law16: Cross-Domain Consistency — no SI without cause */
        bool siOK = !siSignalActive || reactorTripStatus;

        return chemicalOK && ionicOK && thermalOK && mechanicalOK &&
               tripOK && responsive && dnbrOK && marginOK && containmentOK && siOK;
    }

    bool isHarmonious() {
        return computeMu() >= HARMONY_THRESHOLD && checkCH();
    }

    void updateScore(byte index, float newScore) {
        if (index < count) {
            scores[index] = constrain(newScore, 0.0001f, 1.0f);
        }
    }

    void setReactorPower(float p) { reactorPowerPercent = constrain(p, 0.0f, 125.0f); }
    void setDNBR(float d) { dnbr = constrain(d, 0.5f, 5.0f); }
    void setContainmentPressure(float p) { containmentPressureKpa = constrain(p, 50.0f, 1000.0f); }
    void setReactorTrip(bool t) { reactorTripStatus = t; }
    void setSISignal(bool s) { siSignalActive = s; }
    float getDomainMu(byte domain) { return (domain < DOMAIN_COUNT) ? domainMu[domain] : 0.0f; }
    float getReactorPower() { return reactorPowerPercent; }
    float getDNBR() { return dnbr; }
};
